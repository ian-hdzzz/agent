Create Zod schema for MCP server configuration following the pattern in agent-schema.ts.

**mcp-server-schema.ts:**
```typescript
import { z } from 'zod';

export const mcpServerSchema = z.object({
  name: z.string().min(1, 'Name is required'),
  serverType: z.enum(['stdio', 'sse', 'http']),
  // Stdio fields
  command: z.string().optional(),
  args: z.array(z.string()).optional(),
  // SSE/HTTP fields
  url: z.string().url().optional(),
  // Common fields
  env: z.record(z.string()).optional(),
  headers: z.record(z.string()).optional(),
}).refine(
  (data) => {
    if (data.serverType === 'stdio') return !!data.command;
    if (data.serverType === 'sse' || data.serverType === 'http') return !!data.url;
    return true;
  },
  { message: 'Command required for stdio, URL required for sse/http' }
);

export type McpServerFormValues = z.infer<typeof mcpServerSchema>;
```
