1. Create `knowledge-schema.ts` following mcp-server-schema pattern:
   ```typescript
   import { z } from 'zod';

   export const knowledgeSchema = z.object({
     name: z.string().min(1, 'Knowledge base name is required'),
     sourceType: z.enum(['document', 'url']),
     sources: z.array(z.string()).min(1, 'At least one source is required'),
     chunkSize: z.number().min(100).max(10000).optional(),
     overlapSize: z.number().min(0).max(1000).optional(),
   });

   export type KnowledgeFormValues = z.infer<typeof knowledgeSchema>;

   export const defaultKnowledgeValues: Partial<KnowledgeFormValues> = {
     name: '',
     sourceType: 'document',
     sources: [],
     chunkSize: 1000,
     overlapSize: 200,
   };
   ```

2. Create `knowledge-form.tsx` following mcp-server-form pattern:

     - For 'url': Input fields with placeholder "URL (e.g., https://docs.example.com)"
     - Add Source button, remove source (X) button for each
   - Advanced section (collapsible) with chunkSize and overlapSize sliders
