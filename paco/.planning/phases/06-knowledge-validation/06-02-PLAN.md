  <action>
1. Create `logic-schema.ts`:
   ```typescript
   import { z } from 'zod';

   export const logicSchema = z.object({
     name: z.string().min(1, 'Logic node name is required'),
     logicType: z.enum(['condition', 'loop', 'parallel', 'approval']),
     condition: z.string().optional(),
     maxIterations: z.number().min(1).max(100).optional(),
   }).refine(
     (data) => {
       // Condition required for condition and loop types
       if (data.logicType === 'condition' || data.logicType === 'loop') {
         return !!data.condition && data.condition.trim().length > 0;
       }
       return true;
     },
     {
       message: 'Condition expression is required for this logic type',
       path: ['condition'],
     }
   );

   export type LogicFormValues = z.infer<typeof logicSchema>;

   export const defaultLogicValues: Partial<LogicFormValues> = {
     name: '',
     logicType: 'condition',
     condition: '',
     maxIterations: 10,
   };
   ```
