// paco/frontend/types/datamodel.ts
'use client';

// ============================================
// PACO Component System
// ============================================

// Component type discriminator
export type ComponentTypes = 'agent' | 'mcp-server' | 'tool' | 'knowledge' | 'logic' | 'workflow';

// Base component wrapper (similar to AutoGen but with PACO types)
export interface Component<T extends ComponentConfig> {
  id: string;
  type: ComponentTypes;
  label: string;
  description?: string;
  config: T;
}

// Union of all config types
export type ComponentConfig =
  | AgentConfig
  | McpServerConfig
  | ToolConfig
  | KnowledgeConfig
  | LogicConfig
  | WorkflowConfig;

// ============================================
// Agent Configuration (Claude Agent SDK aligned)
// ============================================

export type ClaudeModel = 'claude-sonnet-4-20250514' | 'claude-opus-4-20250514' | 'claude-haiku-3-20240307';

export interface AgentConfig {
  name: string;
  description: string;
  systemPrompt: string;
  model: ClaudeModel;
  temperature?: number;         // 0-1, default 1
  maxTokens?: number;           // Default 4096
  tools: string[];              // Built-in tool names
  mcpServers: string[];         // References to MCP server component IDs
}

// ============================================
// MCP Server Configuration
// ============================================

export type McpServerType = 'stdio' | 'sse' | 'http';

export interface McpServerConfig {
  name: string;
  serverType: McpServerType;
  // For stdio type
  command?: string;
  args?: string[];
  // For sse/http types
  url?: string;
  headers?: Record<string, string>;
  // Common
  env?: Record<string, string>;
  enabledTools?: string[];      // Subset of tools to enable (empty = all)
}

// ============================================
// Tool Configuration (Built-in Claude tools)
// ============================================

export type BuiltInToolName =
  | 'computer'
  | 'bash'
  | 'file_editor'
  | 'web_search'
  | 'mcp';  // MCP is special - connects to MCP servers

export interface ToolConfig {
  name: string;
  toolType: BuiltInToolName;
  enabled: boolean;
  // Tool-specific settings
  settings?: Record<string, unknown>;
}

// ============================================
// Knowledge Base Configuration (Future)
// ============================================

export type KnowledgeSourceType = 'document' | 'url' | 'database';

export interface KnowledgeConfig {
  name: string;
  sourceType: KnowledgeSourceType;
  sources: string[];            // File paths, URLs, or connection strings
  chunkSize?: number;
  overlapSize?: number;
}

// ============================================
// Logic Node Configuration (Future)
// ============================================

export type LogicType = 'condition' | 'loop' | 'parallel' | 'approval';

export interface LogicConfig {
  name: string;
  logicType: LogicType;
  condition?: string;           // For condition type
  maxIterations?: number;       // For loop type
}

// ============================================
// Workflow Configuration (Top-level container)
// ============================================

export interface WorkflowConfig {
  name: string;
  description: string;
  version: string;
  agents: string[];             // Agent component IDs
  entryPoint: string;           // ID of starting agent
}

// ============================================
// Legacy Compatibility Types
// These maintain compatibility with AutoGen Studio store
// while using PACO semantics
// ============================================

// Alias for backward compatibility with store.tsx
export type TeamConfig = WorkflowConfig;
export type WorkbenchConfig = McpServerConfig;
export type StaticWorkbenchConfig = McpServerConfig;

// ============================================
// Canvas Node/Edge Types
// ============================================

export interface PacoNodeData {
  component: Component<ComponentConfig>;
  isUserPositioned?: boolean;
}

export interface PacoEdgeData {
  connectionType: 'tool' | 'mcp' | 'knowledge' | 'flow';
}
```

// paco/frontend/types/guards.ts
'use client';

import {
  Component,
  ComponentConfig,
  ComponentTypes,
  AgentConfig,
  McpServerConfig,
  ToolConfig,
  KnowledgeConfig,
  LogicConfig,
  WorkflowConfig,
  TeamConfig,
  WorkbenchConfig,
  StaticWorkbenchConfig,
} from './datamodel';

// ============================================
// Type Guards for PACO Components
// ============================================

// Check component type
export function isAgentComponent(
  component: Component<ComponentConfig>
): component is Component<AgentConfig> {
  return component.type === 'agent';
}

export function isMcpServerComponent(
  component: Component<ComponentConfig>
): component is Component<McpServerConfig> {
  return component.type === 'mcp-server';
}

export function isToolComponent(
  component: Component<ComponentConfig>
): component is Component<ToolConfig> {
  return component.type === 'tool';
}

export function isKnowledgeComponent(
  component: Component<ComponentConfig>
): component is Component<KnowledgeConfig> {
  return component.type === 'knowledge';
}

export function isLogicComponent(
  component: Component<ComponentConfig>
): component is Component<LogicConfig> {
  return component.type === 'logic';
}

export function isWorkflowComponent(
  component: Component<ComponentConfig>
): component is Component<WorkflowConfig> {
  return component.type === 'workflow';
}

// ============================================
// Legacy Compatibility Guards
// These are used by the ported AutoGen Studio store
// Map to PACO equivalents
// ============================================

// Team = Workflow in PACO
export function isTeamComponent(
  component: Component<ComponentConfig>
): component is Component<TeamConfig> {
  return component.type === 'workflow';
}

// Workbench = MCP Server in PACO
export function isWorkbenchComponent(
  component: Component<ComponentConfig>
): component is Component<WorkbenchConfig> {
  return component.type === 'mcp-server';
}

export function isStaticWorkbench(
  component: Component<ComponentConfig>
): component is Component<StaticWorkbenchConfig> {
  return component.type === 'mcp-server';
}

// ============================================
// AutoGen Legacy Guards (return false - not used in PACO)
// These exist only to satisfy imports from store.tsx
// ============================================

export function isTerminationComponent(
  _component: Component<ComponentConfig>
): boolean {
  // PACO doesn't have termination components
  return false;
}

export function isModelComponent(
  _component: Component<ComponentConfig>
): boolean {
  // PACO doesn't have separate model components - model is part of AgentConfig
  return false;
}

export function isSelectorTeam(
  _component: Component<ComponentConfig>
): boolean {
  // AutoGen-specific - not used in PACO
  return false;
}

export function isAssistantAgent(
  component: Component<ComponentConfig>
): boolean {
  // All PACO agents are "assistant" type
  return component.type === 'agent';
}

export function isWebSurferAgent(
  _component: Component<ComponentConfig>
): boolean {
  // AutoGen-specific - not used in PACO
  return false;
}
```
